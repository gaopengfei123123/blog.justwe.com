<div class="mytest">

<p>{{message}}</p>
{{--和模型绑定,这样的话输入框和上面的值能够同步更改--}}
<input v-model="message">
    <ul>
{{--vue中的遍历,和php的遍历相反, items是这个模型中存入的值,todo是遍历出来的单个元素,--}}

<li v-for="todo in items">
{{ todo.text }}
</li>
    </ul>

//        这里是筛选器获取目标class

items: [
                {text:'first'},
{text:'second'},
{text:'third'},
]
        }
    })
</script>
