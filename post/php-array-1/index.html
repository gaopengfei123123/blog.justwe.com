$testArr = [
    ['id'=>1,'msg'=>'aaa'],
    ['id'=>2,'msg'=>'bbb'],
    ['id'=>3,'msg'=>'ccc'],
    ['id'=>4,'msg'=>'ddd'],
    ];
$roler = [2,1,3,4];

usort($testArr,function($a,$b) use($roler){
    return (array_search($a['id'], $roler) < array_search($b['id'], $roler)) ? -1 : 1;
});


var_dump($testArr);
die();
