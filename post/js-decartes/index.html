function descartes(){
    if( arguments.length < 2 ) return arguments[0] || [];

    return [].reduce.call(arguments, function(col, set) {
        var res = [];
        col.forEach(function(c) {set.forEach(function(s) {
            var t = [].concat( Array.isArray(c) ? c : [c] );
            t.push(s);
            res.push(t);
        })});
        return res;
    });
}
